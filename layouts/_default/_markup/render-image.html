{{- /* Image render hook: add intrinsic width/height (prevents layout shift / CLS), loading="lazy" and decoding="async" to content images. Page-bundle images are resolved as page resources so real pixel dimensions are emitted; external or root-relative images fall back to a plain lazy-loaded . */ -}} {{- $dest := .Destination -}} {{- $alt := .PlainText -}} {{- $img := "" -}} {{- if not (or (hasPrefix $dest "http") (hasPrefix $dest "//") (hasPrefix $dest "/")) -}} {{- with .Page.Resources.GetMatch $dest -}} {{- $img = . -}} {{- end -}} {{- end -}} {{- if and $img (gt $img.Width 0) (gt $img.Height 0) -}} {{- else -}} {{- end -}}
